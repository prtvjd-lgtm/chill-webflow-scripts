<!-- Absolute child height sync -->
<script>
(function () {
  if (window.__chillAbsoluteChildHeightSyncStarted) return;
  window.__chillAbsoluteChildHeightSyncStarted = true;

  var PARENT_SELECTOR = '[data-absolute-height-parent]';
  var CHILD_SELECTOR = '[data-absolute-height-child]';
  var HEIGHT_ANIMATION = {
    duration: 600,
    easing: 'cubic-bezier(0.075, 0.82, 0.165, 1)'
  };

  function getTargetHeight(child) {
    return Math.max(0, child.offsetTop + child.offsetHeight);
  }

  function getAutoHeight(parent) {
    var height = parent.style.height;
    var transition = parent.style.transition;
    var overflow = parent.style.overflow;

    parent.style.transition = 'none';
    parent.style.height = 'auto';

    var autoHeight = parent.getBoundingClientRect().height;

    parent.style.height = height;
    parent.style.transition = transition;
    parent.style.overflow = overflow;

    return autoHeight;
  }

  function prefersReducedMotion() {
    return !!(
      window.matchMedia &&
      window.matchMedia('(prefers-reduced-motion: reduce)').matches
    );
  }

  function clearHeightAnimation(parent, restoreStyles) {
    var animation = parent.__chillAbsoluteHeightAnimation;

    if (!animation) return null;

    window.clearTimeout(animation.timer);
    parent.removeEventListener('transitionend', animation.onEnd);

    if (restoreStyles) {
      parent.style.transition = animation.transition;
      parent.style.overflow = animation.overflow;
    }

    parent.__chillAbsoluteHeightAnimation = null;

    return animation;
  }

  function setHeightImmediately(parent, height) {
    clearHeightAnimation(parent, true);
    parent.style.height = height + 'px';
  }

  function animateHeight(parent, targetHeight) {
    if (prefersReducedMotion()) {
      setHeightImmediately(parent, targetHeight);
      return;
    }

    var activeAnimation = parent.__chillAbsoluteHeightAnimation;

    if (
      activeAnimation &&
      Math.abs(activeAnimation.targetHeight - targetHeight) < 0.5
    ) {
      return;
    }

    var startHeight = parent.getBoundingClientRect().height;
    var transition = activeAnimation ? activeAnimation.transition : parent.style.transition;
    var overflow = activeAnimation ? activeAnimation.overflow : parent.style.overflow;

    clearHeightAnimation(parent, false);

    if (Math.abs(startHeight - targetHeight) < 0.5) {
      parent.style.height = targetHeight + 'px';
      parent.style.transition = transition;
      parent.style.overflow = overflow;
      return;
    }

    parent.style.transition = 'none';
    parent.style.height = startHeight + 'px';
    parent.style.overflow = 'hidden';

    parent.getBoundingClientRect();
    parent.style.transition =
      'height ' + HEIGHT_ANIMATION.duration + 'ms ' + HEIGHT_ANIMATION.easing;
    parent.style.height = targetHeight + 'px';

    var animation = {
      targetHeight: targetHeight,
      transition: transition,
      overflow: overflow,
      timer: 0,
      onEnd: null
    };

    function finishAnimation(event) {
      if (event && (event.target !== parent || event.propertyName !== 'height')) return;
      if (parent.__chillAbsoluteHeightAnimation !== animation) return;

      clearHeightAnimation(parent, false);
      parent.style.transition = transition;
      parent.style.overflow = overflow;
    }

    animation.onEnd = finishAnimation;
    animation.timer = window.setTimeout(
      finishAnimation,
      HEIGHT_ANIMATION.duration + 80
    );
    parent.__chillAbsoluteHeightAnimation = animation;
    parent.addEventListener('transitionend', finishAnimation);
  }

  function syncHeight(parent, child, refreshMinimumHeight, animate) {
    var state = parent.__chillAbsoluteHeightSyncState;

    if (!state) {
      state = { initialized: false, minimumHeight: 0 };
      parent.__chillAbsoluteHeightSyncState = state;
    }

    if (refreshMinimumHeight) {
      state.minimumHeight = getAutoHeight(parent);
    }

    var targetHeight = Math.max(getTargetHeight(child), state.minimumHeight);

    if (!state.initialized) {
      state.initialized = true;
      setHeightImmediately(parent, targetHeight);
      return;
    }

    if (animate === false) {
      setHeightImmediately(parent, targetHeight);
      return;
    }

    animateHeight(parent, targetHeight);
  }

  function initParent(parent) {
    if (parent.dataset.absoluteHeightReady === 'true') return;

    var child = parent.querySelector(CHILD_SELECTOR);

    if (!child) return;

    parent.dataset.absoluteHeightReady = 'true';

    var frame = 0;
    var shouldRefreshMinimumHeight = false;
    var shouldSkipAnimation = false;
    var scheduleSync = function (refreshMinimumHeight, animate) {
      shouldRefreshMinimumHeight = shouldRefreshMinimumHeight || !!refreshMinimumHeight;
      shouldSkipAnimation = shouldSkipAnimation || animate === false;

      if (frame) return;

      frame = window.requestAnimationFrame(function () {
        frame = 0;
        var refresh = shouldRefreshMinimumHeight;
        var shouldAnimate = !shouldSkipAnimation;

        shouldRefreshMinimumHeight = false;
        shouldSkipAnimation = false;
        syncHeight(parent, child, refresh, shouldAnimate);
      });
    };

    if (window.ResizeObserver) {
      var resizeObserver = new ResizeObserver(function () {
        scheduleSync(false, true);
      });
      resizeObserver.observe(child);
    }

    window.addEventListener('resize', function () {
      scheduleSync(true, false);
    }, { passive: true });

    if (document.fonts && document.fonts.ready) {
      document.fonts.ready.then(function () {
        scheduleSync(true, false);
      });
    }

    scheduleSync(true, false);
  }

  function initIn(scope) {
    if (!scope || scope.nodeType !== Node.ELEMENT_NODE) return;

    var closestParent = scope.closest(PARENT_SELECTOR);

    if (closestParent) initParent(closestParent);

    if (scope.matches(PARENT_SELECTOR)) initParent(scope);

    Array.prototype.forEach.call(scope.querySelectorAll(PARENT_SELECTOR), initParent);
  }

  function observeNewParents() {
    if (!document.body || !window.MutationObserver) return;

    var observer = new MutationObserver(function (records) {
      records.forEach(function (record) {
        Array.prototype.forEach.call(record.addedNodes, initIn);
      });
    });

    observer.observe(document.body, { childList: true, subtree: true });
  }

  function init() {
    initIn(document.documentElement);
    observeNewParents();
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init, { once: true });
  } else {
    init();
  }
})();
</script>
