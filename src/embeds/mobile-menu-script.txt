<!-- Мобильное меню -->
<style>
  @media (max-width: 990px) {
    [data-mobile-menu]:not(.is-mobile-menu-ready),
    [data-mobile-menu].is-mobile-menu-closed {
      display: none !important;
    }
  }
</style>

<script>
(function () {
  if (window.WebflowMobileMenu) return;

  window.WebflowMobileMenu = true;

  const SELECTORS = {
    toggle: '[data-mobile-menu-toggle]',
    menu: '[data-mobile-menu]',
    item: '[data-mobile-menu-item]'
  };

  const CLOSED_CLASS = 'is-mobile-menu-closed';
  const READY_CLASS = 'is-mobile-menu-ready';
  const ANIMATION = {
    duration: 600,
    stagger: 70,
    offsetX: 10,
    easing: 'cubic-bezier(0.075, 0.82, 0.165, 1)'
  };

  const compactViewport = window.matchMedia('(max-width: 990px)');
  const usedMenus = new Set();
  const states = Array.from(document.querySelectorAll(SELECTORS.toggle))
    .map(function (toggle, index) {
      const header = toggle.closest('.header');
      const menu = header
        ? header.querySelector(SELECTORS.menu)
        : document.querySelector(SELECTORS.menu);

      if (!menu || usedMenus.has(menu)) return null;

      usedMenus.add(menu);

      if (!menu.id) {
        menu.id = 'mobile-menu-' + (index + 1);
      }

      toggle.setAttribute('aria-controls', menu.id);

      return {
        toggle: toggle,
        menu: menu,
        items: Array.from(menu.querySelectorAll(SELECTORS.item)),
        itemStyles: new WeakMap(),
        initialAttributes: {
          menuAriaHidden: menu.getAttribute('aria-hidden'),
          toggleAriaExpanded: toggle.getAttribute('aria-expanded')
        },
        timers: [],
        revealFrame: 0
      };
    })
    .filter(Boolean);

  if (!states.length) return;

  function prefersReducedMotion() {
    return window.matchMedia && window.matchMedia('(prefers-reduced-motion: reduce)').matches;
  }

  function restoreAttribute(element, name, value) {
    if (value === null) {
      element.removeAttribute(name);
      return;
    }

    element.setAttribute(name, value);
  }

  function saveItemStyles(state, item) {
    if (state.itemStyles.has(item)) {
      return state.itemStyles.get(item);
    }

    const styles = {
      opacity: item.style.opacity,
      transform: item.style.transform,
      transition: item.style.transition,
      willChange: item.style.willChange,
      baseTransform: ''
    };

    state.itemStyles.set(item, styles);
    return styles;
  }

  function restoreItemStyles(state, item) {
    const styles = state.itemStyles.get(item);

    if (!styles) return;

    item.style.opacity = styles.opacity;
    item.style.transform = styles.transform;
    item.style.transition = styles.transition;
    item.style.willChange = styles.willChange;
    state.itemStyles.delete(item);
  }

  function clearAnimation(state) {
    window.cancelAnimationFrame(state.revealFrame);
    state.revealFrame = 0;

    state.timers.forEach(function (timer) {
      window.clearTimeout(timer);
    });

    state.timers = [];
    state.items.forEach(function (item) {
      restoreItemStyles(state, item);
    });
  }

  function clearAllAnimations() {
    states.forEach(clearAnimation);
  }

  function closeMenus() {
    if (!compactViewport.matches) return;

    clearAllAnimations();

    states.forEach(function (state) {
      state.menu.classList.add(CLOSED_CLASS);
      state.menu.setAttribute('aria-hidden', 'true');
      state.toggle.setAttribute('aria-expanded', 'false');
    });
  }

  function restoreDesktopMenus() {
    clearAllAnimations();

    states.forEach(function (state) {
      state.menu.classList.remove(CLOSED_CLASS);
      state.menu.hidden = false;
      restoreAttribute(state.menu, 'aria-hidden', state.initialAttributes.menuAriaHidden);
      restoreAttribute(state.toggle, 'aria-expanded', state.initialAttributes.toggleAriaExpanded);
    });
  }

  function prepareItem(state, item) {
    const styles = saveItemStyles(state, item);
    const computedTransform = window.getComputedStyle(item).transform;

    styles.baseTransform = computedTransform === 'none' ? '' : computedTransform;
    item.style.opacity = '0';
    item.style.transform = 'translate3d(' + ANIMATION.offsetX + 'px, 0, 0) ' + styles.baseTransform;
    item.style.transition = 'none';
    item.style.willChange = 'opacity, transform';
  }

  function animateItems(state) {
    state.revealFrame = window.requestAnimationFrame(function () {
      state.items.forEach(function (item, index) {
        const timer = window.setTimeout(function () {
          const styles = state.itemStyles.get(item);

          if (!styles) return;

          item.style.transition =
            'opacity ' + ANIMATION.duration + 'ms ' + ANIMATION.easing + ', ' +
            'transform ' + ANIMATION.duration + 'ms ' + ANIMATION.easing;
          item.style.opacity = '1';
          item.style.transform = styles.baseTransform;
        }, index * ANIMATION.stagger);

        state.timers.push(timer);
      });

      const cleanupTimer = window.setTimeout(function () {
        state.items.forEach(function (item) {
          restoreItemStyles(state, item);
        });
        state.timers = [];
      }, Math.max(0, state.items.length - 1) * ANIMATION.stagger + ANIMATION.duration + 80);

      state.timers.push(cleanupTimer);
    });
  }

  function openMenus() {
    if (!compactViewport.matches) return;

    clearAllAnimations();

    states.forEach(function (state) {
      state.menu.hidden = false;
      state.menu.classList.remove(CLOSED_CLASS);
      state.menu.setAttribute('aria-hidden', 'false');
      state.toggle.setAttribute('aria-expanded', 'true');
    });

    if (prefersReducedMotion()) return;

    states.forEach(function (state) {
      state.items.forEach(function (item) {
        prepareItem(state, item);
      });
      state.menu.getBoundingClientRect();
      animateItems(state);
    });
  }

  function menusAreOpen() {
    return states.some(function (state) {
      return !state.menu.classList.contains(CLOSED_CLASS);
    });
  }

  function syncViewport() {
    if (compactViewport.matches) {
      closeMenus();
      return;
    }

    restoreDesktopMenus();
  }

  states.forEach(function (state) {
    state.toggle.addEventListener('click', function (event) {
      if (!compactViewport.matches) return;

      event.preventDefault();

      if (menusAreOpen()) {
        closeMenus();
      } else {
        openMenus();
      }
    });
  });

  if (compactViewport.addEventListener) {
    compactViewport.addEventListener('change', syncViewport);
  } else {
    compactViewport.addListener(syncViewport);
  }

  syncViewport();

  states.forEach(function (state) {
    state.menu.classList.add(READY_CLASS);
  });
})();
</script>
