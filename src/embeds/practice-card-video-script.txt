<!-- Видео в карточках практик -->
<style>
.practice-hover-card[data-practice-video-url]:not([data-practice-video-url=""]):not(.is-practice-video-error) > .image,
.practice-hover-card[data-practice-video-url]:not([data-practice-video-url=""]):not(.is-practice-video-error) > .image-wrapper > .image,
.practice-hover-card[data-practice-video-url]:not([data-practice-video-url=""]):not(.is-practice-video-error) > .practice-card-media > .image {
  visibility: hidden !important;
}

.practice-card-media {
  position: relative;
  width: 100%;
  border-radius: 10px;
  overflow: hidden;
}

.practice-card-media > .image {
  display: block;
  width: 100%;
}

.image-wrapper[data-practice-card-media] {
  position: relative;
}

.practice-card-media > .practice-card-video,
.image-wrapper[data-practice-card-media] > .practice-card-video {
  position: absolute;
  inset: 0;
  display: block;
  width: 100%;
  height: 100%;
  object-fit: cover;
  opacity: 0;
  pointer-events: none;
}

.practice-hover-card.is-practice-video-ready .practice-card-media > .practice-card-video,
.practice-hover-card.is-practice-video-ready .image-wrapper[data-practice-card-media] > .practice-card-video {
  opacity: 1;
}
</style>

<script>
(function () {
  if (window.__chillPracticeCardVideosStarted) return;
  window.__chillPracticeCardVideosStarted = true;

  const CARD_SELECTOR = '[data-practice-card]';
  const HOVER_CARD_SELECTOR = '.practice-hover-card';
  const VIDEO_ERROR_CLASS = 'is-practice-video-error';
  const VIDEO_READY_CLASS = 'is-practice-video-ready';
  const reducedMotion = window.matchMedia && window.matchMedia('(prefers-reduced-motion: reduce)');
  let states = [];

  function getSafeVideoUrl(value) {
    if (!value) return '';

    try {
      const url = new URL(value, window.location.origin);
      const host = url.hostname.toLowerCase();
      const isYandexStorage = host === 'storage.yandexcloud.net' || host.endsWith('.storage.yandexcloud.net');

      if (url.protocol !== 'https:' || !isYandexStorage || !url.pathname.toLowerCase().endsWith('.mp4')) {
        return '';
      }

      return url.href;
    } catch (error) {
      return '';
    }
  }

  function createState(card) {
    const hoverCard = card.querySelector(HOVER_CARD_SELECTOR);
    if (!hoverCard) return null;

    const source = getSafeVideoUrl(hoverCard.getAttribute('data-practice-video-url'));

    if (!source) {
      if (hoverCard.getAttribute('data-practice-video-url')) {
        hoverCard.classList.add(VIDEO_ERROR_CLASS);
      }
      return null;
    }

    const imageWrapper = hoverCard.querySelector(':scope > .image-wrapper');
    const image = imageWrapper
      ? imageWrapper.querySelector(':scope > .image')
      : hoverCard.querySelector(':scope > .image');
    const media = imageWrapper || document.createElement('div');
    const video = document.createElement('video');

    if (imageWrapper) {
      media.setAttribute('data-practice-card-media', '');
    } else {
      media.className = 'practice-card-media';
    }

    video.className = 'practice-card-video';
    video.src = source;
    video.muted = true;
    video.defaultMuted = true;
    video.loop = true;
    video.playsInline = true;
    video.preload = 'metadata';
    video.setAttribute('aria-hidden', 'true');
    video.setAttribute('tabindex', '-1');

    video.addEventListener('loadeddata', function () {
      hoverCard.classList.add(VIDEO_READY_CLASS);
      window.dispatchEvent(new Event('resize'));
    });

    video.addEventListener('error', function () {
      hoverCard.classList.remove(VIDEO_READY_CLASS);
      hoverCard.classList.add(VIDEO_ERROR_CLASS);
      video.remove();
    }, { once: true });

    if (imageWrapper) {
      if (image) {
        image.loading = 'eager';
      }
    } else if (image) {
      image.loading = 'eager';
      image.insertAdjacentElement('beforebegin', media);
      media.appendChild(image);
    } else {
      hoverCard.insertBefore(media, hoverCard.firstChild);
    }

    media.appendChild(video);

    return {
      card: card,
      hoverCard: hoverCard,
      video: video,
      preloadRequested: false
    };
  }

  function isVisible(state) {
    return state.card.getClientRects().length > 0 && window.getComputedStyle(state.card).display !== 'none';
  }

  function preloadVisibleVideos() {
    states.forEach(function (state) {
      if (!isVisible(state) || state.preloadRequested) return;

      state.preloadRequested = true;
      state.video.preload = 'auto';
      state.video.load();
    });
  }

  function pauseAll(exceptState) {
    states.forEach(function (state) {
      if (state === exceptState) return;
      state.video.pause();
    });
  }

  function play(state) {
    pauseAll(state);

    if (document.hidden || (reducedMotion && reducedMotion.matches)) return;

    const playPromise = state.video.play();
    if (playPromise && typeof playPromise.catch === 'function') {
      playPromise.catch(function () {});
    }
  }

  function init() {
    states = Array.from(document.querySelectorAll(CARD_SELECTOR))
      .map(createState)
      .filter(Boolean);

    states.forEach(function (state) {
      state.card.addEventListener('mouseenter', function () {
        play(state);
      });

      state.card.addEventListener('mouseleave', function () {
        state.video.pause();
      });
    });

    document.addEventListener('visibilitychange', function () {
      if (document.hidden) pauseAll();
    });

    if (reducedMotion && typeof reducedMotion.addEventListener === 'function') {
      reducedMotion.addEventListener('change', function () {
        if (reducedMotion.matches) pauseAll();
      });
    }

    window.addEventListener('practice-hover-cards-changed', preloadVisibleVideos);
    preloadVisibleVideos();
  }

  if (window.Webflow && typeof window.Webflow.push === 'function') {
    window.Webflow.push(init);
  } else if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
</script>
