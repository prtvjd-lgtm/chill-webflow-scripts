<!-- Plain text paragraphs -->
<style>
.chill-plain-text-paragraph {
  display: block;
  margin: 0 !important;
  white-space: pre-line;
}

@supports (margin-top: 1lh) {
  .chill-plain-text-paragraph + .chill-plain-text-paragraph {
    margin-top: 0.64lh !important;
  }
}
</style>

<script>
(function () {
  if (window.__chillCmsPlainTextParagraphsStarted) return;
  window.__chillCmsPlainTextParagraphsStarted = true;

  var PARAGRAPH_CLASS = 'chill-plain-text-paragraph';
  var PARAGRAPH_SELECTOR = '.' + PARAGRAPH_CLASS;
  var SKIPPED_TAGS = new Set(['SCRIPT', 'STYLE', 'NOSCRIPT', 'TEMPLATE', 'TEXTAREA', 'OPTION']);
  var supportsLineHeightUnit = !!(window.CSS && window.CSS.supports && window.CSS.supports('margin-top', '1lh'));

  function splitParagraphs(text) {
    return String(text || '')
      .replace(/\r\n?/g, '\n')
      .split(/\n[ \t\u00a0]*\n+/)
      .map(function (paragraph) {
        return paragraph.trim();
      })
      .filter(Boolean);
  }

  function hasLineBreak(text) {
    return /\r\n?|\n/.test(String(text || ''));
  }

  function isPlainTextElement(element) {
    if (!element || element.nodeType !== Node.ELEMENT_NODE) return false;
    if (SKIPPED_TAGS.has(element.tagName) || element.classList.contains(PARAGRAPH_CLASS)) return false;
    if (element.children.length || element.closest('[contenteditable="true"]')) return false;

    return hasLineBreak(element.textContent);
  }

  function prepareElement(element) {
    if (!isPlainTextElement(element)) return;

    var paragraphs = splitParagraphs(element.textContent);
    var fragment = document.createDocumentFragment();

    paragraphs.forEach(function (text) {
      var paragraph = document.createElement('span');
      paragraph.className = PARAGRAPH_CLASS;
      paragraph.textContent = text;
      fragment.appendChild(paragraph);
    });

    element.replaceChildren(fragment);
    setFallbackMargins(element);
  }

  function getLineHeight(element) {
    var styles = window.getComputedStyle(element);
    var lineHeight = parseFloat(styles.lineHeight);

    if (Number.isFinite(lineHeight)) return lineHeight;

    return parseFloat(styles.fontSize) * 1.2;
  }

  function setFallbackMargins(scope) {
    if (supportsLineHeightUnit) return;

    var paragraphs = scope.querySelectorAll(PARAGRAPH_SELECTOR);

    Array.prototype.forEach.call(paragraphs, function (paragraph, index) {
      if (index === 0) return;

      paragraph.style.setProperty('margin-top', (getLineHeight(paragraph) * 0.64) + 'px', 'important');
    });
  }

  function prepareIn(scope) {
    if (!scope || scope.nodeType !== Node.ELEMENT_NODE) return;

    prepareElement(scope);
    Array.prototype.forEach.call(scope.querySelectorAll('*'), prepareElement);
    setFallbackMargins(scope);
  }

  function updateFallbackMargins() {
    if (!supportsLineHeightUnit) setFallbackMargins(document.documentElement);
  }

  function observeNewRoots() {
    if (!document.body || !window.MutationObserver) return;

    var observer = new MutationObserver(function (records) {
      records.forEach(function (record) {
        Array.prototype.forEach.call(record.addedNodes, prepareIn);
      });
    });

    observer.observe(document.body, { childList: true, subtree: true });
  }

  function init() {
    prepareIn(document.documentElement);
    observeNewRoots();

    if (!supportsLineHeightUnit) {
      window.addEventListener('resize', updateFallbackMargins, { passive: true });
      if (document.fonts && document.fonts.ready) document.fonts.ready.then(updateFallbackMargins);
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init, { once: true });
  } else {
    init();
  }
})();
</script>
